unsigned long long cgc_promptc(unsigned long long a0, unsigned short a1, unsigned long long a2)
{
    unsigned long long count;  // rax

    if (cgc_sendall(1, a2, cgc_strlen(a2)) < 0)
        cgc__terminate(3);
    count = cgc_recvline(0, a0, a1);
    if (count >= 0)
        return count;
    if (cgc_sendline(1, "You didn't say the magic word.", 30) >= 0)
        return cgc__terminate(2);
    cgc__terminate(6);
    return cgc__terminate(2);
}
